       ```
       (function(){
         'use strict';
         const SKELETON_DELAY_MS = 300;
         const timers = new WeakMap();
         const originals = new WeakMap();

         function inferVariant(target){
           return target.getAttribute('data-skeleton') || 'text';
         }
         function buildSkeleton(variant, target){
           const count = target.getAttribute('data-skeleton-count') || '3';
           return `<ag-skeleton variant="${variant}" count="${count}"></ag-skeleton>`;
         }
         document.body.addEventListener('htmx:beforeRequest', function(evt){
           const target = evt.detail.target;
           if (!target || target === document.body) return;
           const t = setTimeout(() => {
             originals.set(target, target.innerHTML);
             target.innerHTML = buildSkeleton(inferVariant(target), target);
             target.setAttribute('aria-busy', 'true');
           }, SKELETON_DELAY_MS);
           timers.set(target, t);
         });
         document.body.addEventListener('htmx:afterRequest', function(evt){
           const target = evt.detail.target;
           if (!target) return;
           const t = timers.get(target);
           if (t) clearTimeout(t);
           timers.delete(target);
           target.removeAttribute('aria-busy');
           // If failure AND we already swapped to skeleton (rare), restore original
           if (!evt.detail.successful && originals.has(target)) {
             target.innerHTML = originals.get(target);
           }
           originals.delete(target);
         });

         // Exported helper for fetch-based optimistic flows (used by Task 3)
         window.LoadingStates = {
           SKELETON_DELAY_MS,
           applyOptimistic(el, mutate, request, rollback){
             // mutate() : DOM update sync ; request() : Promise ; rollback(err) : on reject
             el.classList.add('is-pending');
             try { mutate(); } catch(e){ el.classList.remove('is-pending'); throw e; }
             return request()
               .then(function(res){ el.classList.remove('is-pending'); return res; })
               .catch(function(err){ el.classList.remove('is-pending'); rollback(err); throw err; });
           }
         };
       })();
       ```
       Ajouter aussi dans `design-system.css` `.is-pending { opacity: 0.6; transition: opacity var(--duration-fast) var(--ease-default); }` avec `@media (prefers-reduced-motion) .is-pending { opacity: 0.7; transition: none; }`.
