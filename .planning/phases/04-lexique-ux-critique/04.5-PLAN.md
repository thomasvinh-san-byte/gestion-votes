```php
<?php

declare(strict_types=1);

namespace Tests\Security;

use PHPUnit\Framework\TestCase;

/**
 * UX conventions sur ErrorDictionary (ERR-02 + ERR-03).
 *
 * Filet permanent : tout message d'erreur des 50 codes les plus émis
 * doit contenir un next-step actionnable (virgule + verbe d'action).
 * Aucun message ne doit contenir une phrase creuse interdite.
 *
 * Note : ne duplique PAS CopyConventionsTest (forbidden words copropriété/syndic).
 */
final class UxConventionsTest extends TestCase
{
    /**
     * Top 50 codes émis (snapshot Phase 4 / plan 04.4).
     * À réviser 1× par milestone si le ranking change.
     */
    private const TOP_50_CODES = [
        // Liste à figer depuis le SUMMARY 04.4 — exemple :
        'unauthorized',
        'forbidden',
        'invalid_request',
        'meeting_not_found',
        'missing_meeting_id',
        'invalid_meeting_id',
        'motion_not_found',
        'business_error',
        'already_voted',
        'quorum_not_met',
        'rate_limited',
        'invalid_csrf',
        'session_expired',
        'internal_error',
        'not_found',
        // ... (compléter jusqu'à 50 depuis le SUMMARY 04.4)
    ];

    /**
     * Verbes d'action acceptables comme next-step.
     * Liste élargissable mais doit rester en français.
     */
    private const ACTION_VERBS = [
        'demandez', 'contactez', 'vérifiez', 'essayez', 'recommencez',
        'revenez', 'reconnectez', 'attendez', 'consultez', 'accédez',
        'sélectionnez', 'cliquez', 'corrigez', 'complétez', 'fermez',
        'rechargez', 'rafraîchissez', 'patientez', 'choisissez',
        'modifiez', 'mettez', 'saisissez', 'utilisez', 'téléchargez',
        // Subjonctifs
        'que ', // "demandez à l'opérateur que ..."
    ];

    /**
     * Phrases creuses interdites (ERR-03).
     */
    private const FORBIDDEN_PATTERNS = [
        '/réessayer\.?$/i',
        '/réessayez\.?$/i',
        '/contactez (le|l\')admin/i',
        '/erreur survenue/i',
        '/une erreur est survenue/i',
        '/veuillez réessayer plus tard/i',
    ];

    /**
     * Charger toutes les paires code => message du dictionnaire.
     * Stratégie pragmatique : parse texte brut.
     *
     * @return array<string, string>
     */
    private function loadMessages(): array
    {
        $path = dirname(__DIR__, 2) . '/app/Services/ErrorDictionary.php';
        $this->assertFileExists($path);
        $src = file_get_contents($path);
        $this->assertNotFalse($src);

        // Capture les paires "'code' => 'message'," (single ou double quotes)
        $messages = [];
        if (preg_match_all(
            '/[\'"]([a-z][a-z0-9_]*)[\'"]\s*=>\s*[\'"]((?:\\\\.|[^\'"\\\\])*)[\'"]/i',
            $src,
            $m,
            PREG_SET_ORDER
        )) {
            foreach ($m as $pair) {
                $messages[$pair[1]] = $pair[2];
            }
        }
        $this->assertNotEmpty($messages, 'Aucune paire code=>message trouvée dans ErrorDictionary');
        return $messages;
    }

    public function testNextStepPresence(): void
    {
        $messages = $this->loadMessages();
        $missing = [];

        foreach (self::TOP_50_CODES as $code) {
            if (!isset($messages[$code])) {
                $missing[] = "$code (code absent du dictionnaire)";
                continue;
            }
            $msg = $messages[$code];

            // Règle 1 : doit contenir au moins une virgule
            if (strpos($msg, ',') === false) {
                $missing[] = "$code (pas de virgule) : $msg";
                continue;
            }

            // Règle 2 : doit contenir au moins un verbe d'action de la liste blanche
            $msgLower = mb_strtolower($msg, 'UTF-8');
            $hasVerb = false;
            foreach (self::ACTION_VERBS as $verb) {
                if (strpos($msgLower, mb_strtolower($verb, 'UTF-8')) !== false) {
                    $hasVerb = true;
                    break;
                }
            }
            if (!$hasVerb) {
                $missing[] = "$code (pas de verbe d'action) : $msg";
            }
        }

        $this->assertSame(
            [],
            $missing,
            "Codes du top 50 sans next-step actionnable :\n" . implode("\n", $missing)
        );
    }

    public function testNoForbiddenPhrases(): void
    {
        $messages = $this->loadMessages();
        $hits = [];

        foreach ($messages as $code => $msg) {
            foreach (self::FORBIDDEN_PATTERNS as $pattern) {
                if (preg_match($pattern, $msg)) {
                    $hits[] = "$code matche $pattern : $msg";
                }
            }
        }

        $this->assertSame(
            [],
            $hits,
            "Messages contenant une phrase creuse interdite (ERR-03) :\n" . implode("\n", $hits)
        );
    }
}
```
