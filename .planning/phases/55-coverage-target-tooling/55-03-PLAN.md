From app/Core/Providers/RepositoryFactory.php:
```php
final class RepositoryFactory {
    private static ?self $instance = null;
    private ?PDO $pdo;
    private array $cache = [];

    public function __construct(?PDO $pdo = null) { $this->pdo = $pdo; }
    public static function getInstance(): self { return self::$instance ??= new self(); }
    public static function reset(): void { self::$instance = null; }

    // Repository accessors (30+ methods):
    public function agenda(): AgendaRepository { return $this->get(AgendaRepository::class); }
    public function meeting(): MeetingRepository { return $this->get(MeetingRepository::class); }
    // ... etc
    private function get(string $class): object { return $this->cache[$class] ??= new $class($this->pdo); }
}
```

From app/Controller/AbstractController.php:
```php
abstract class AbstractController {
    protected Request $request;
    public function __construct() { $this->request = new Request(); }
    protected function repo(): RepositoryFactory { return RepositoryFactory::getInstance(); }
    public function handle(string $method): void {
        try { $this->$method(); }
        catch (ApiResponseException $e) { throw $e; }
        catch (InvalidArgumentException $e) { api_fail('invalid_request', 422, ...); }
        catch (RuntimeException $e) { api_fail('business_error', 400, ...); }
        catch (Throwable $e) { api_fail('internal_error', 500, ...); }
    }
}
```
