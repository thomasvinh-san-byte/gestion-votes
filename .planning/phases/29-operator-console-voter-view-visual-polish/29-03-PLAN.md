**6. Implement castVoteOptimistic() in vote.js:**
Replace or augment the existing confirmation overlay flow with an optimistic pattern:
```javascript
function castVoteOptimistic(choice) {
  // 1. Instant visual (<50ms — synchronous DOM)
  setVoteSelected(choice);

  // 2. Background submit
  var body = { /* existing cast_vote payload with choice */ };
  window.api('/api/v1/cast_vote.php', { method: 'POST', body: body })
    .then(function(res) {
      if (res.body && res.body.ok) {
        showConfirmationState();
      } else {
        rollbackVote(choice);
        showInlineError('Vote non enregistre -- reessayez');
      }
    })
    .catch(function() {
      rollbackVote(choice);
      showInlineError('Vote non enregistre -- reessayez');
    });
}

function setVoteSelected(choice) {
  // Add .vote-btn-selected class to the tapped button
  // Remove from all others
  document.querySelectorAll('.vote-btn').forEach(function(btn) {
    btn.classList.toggle('vote-btn-selected', btn.dataset.choice === choice);
    btn.disabled = true; // prevent double-tap
  });
}

function rollbackVote(choice) {
  // Remove .vote-btn-selected, re-enable buttons
  document.querySelectorAll('.vote-btn').forEach(function(btn) {
    btn.classList.remove('vote-btn-selected');
    btn.disabled = false;
  });
}

function showConfirmationState() {
  setVoteAppState('confirmed');
  setTimeout(function() { setVoteAppState('waiting'); }, 3000);
}

function showInlineError(msg) {
  // Show error via ag-toast or inline element
  if (window.AgToast) AgToast.error(msg);
}
```

Wire the existing vote buttons to call castVoteOptimistic(choice) instead of the existing confirmation dialog flow. Find where vote buttons currently have click handlers and replace with the optimistic flow.
