```javascript
function loadDocBadges(motions, meetingId) {
  // Fetch all documents for this meeting in one call
  window.api('/api/v1/resolution_documents?motion_id=all&meeting_id=' + meetingId).catch(function() {
    // Fallback: fetch per-motion if bulk endpoint not available
    return null;
  });

  // Alternative: fetch per motion
  motions.forEach(function(motion) {
    window.api('/api/v1/resolution_documents?motion_id=' + motion.id).then(function(resp) {
      renderDocBadge(motion.id, resp.documents ? resp.documents.length : 0);
    }).catch(function() {
      renderDocBadge(motion.id, 0);
    });
  });
}

function renderDocBadge(motionId, docCount) {
  // Find badge elements for this motion
  var badges = document.querySelectorAll('[data-motion-doc-badge][data-motion-id="' + motionId + '"]');
  badges.forEach(function(badge) {
    if (docCount === 0) {
      badge.textContent = 'Aucun document';
      badge.className = 'doc-badge doc-badge--empty';
    } else {
      badge.textContent = docCount + ' document' + (docCount > 1 ? 's joints' : ' joint');
      badge.className = 'doc-badge doc-badge--has-docs';
      badge.style.cursor = 'pointer';
      badge.onclick = function() { openDocViewer(motionId); };
    }
  });
}

function openDocViewer(motionId) {
  // Fetch docs for this motion, then open viewer with first doc
  window.api('/api/v1/resolution_documents?motion_id=' + motionId).then(function(resp) {
    if (!resp.documents || resp.documents.length === 0) return;
    var viewer = document.querySelector('ag-pdf-viewer') || document.createElement('ag-pdf-viewer');
    if (!viewer.parentElement) {
      viewer.setAttribute('mode', 'panel');
      viewer.setAttribute('allow-download', '');
      document.body.appendChild(viewer);
    }
    var doc = resp.documents[0];
    viewer.setAttribute('src', '/api/v1/resolution_document_serve?id=' + doc.id);
    viewer.setAttribute('filename', doc.original_name);
    viewer.open();
  });
}
```

```javascript
function addDocUploadToMotionCard(cardEl, motionId, meetingId) {
  var uploadBtn = document.createElement('button');
  uploadBtn.className = 'btn btn--ghost btn--sm';
  uploadBtn.textContent = '+ Document';
  uploadBtn.title = 'Ajouter un document PDF';

  uploadBtn.addEventListener('click', function() {
    var input = document.createElement('input');
    input.type = 'file';
    input.accept = 'application/pdf';
    input.addEventListener('change', function() {
      if (!input.files || !input.files[0]) return;
      var file = input.files[0];

      if (file.size > 10 * 1024 * 1024) {
        window.AgToast?.show('Le fichier depasse 10 Mo', 'error');
        return;
      }
      if (file.type !== 'application/pdf') {
        window.AgToast?.show('Seuls les fichiers PDF sont acceptes', 'error');
        return;
      }

      var formData = new FormData();
      formData.append('filepond', file);
      formData.append('motion_id', motionId);
      formData.append('meeting_id', meetingId);

      fetch('/api/v1/resolution_documents', {
        method: 'POST',
        headers: {
          'X-Requested-With': 'XMLHttpRequest',
          'X-CSRF-Token': document.querySelector('meta[name="csrf-token"]')?.content || ''
        },
        body: formData
      }).then(function(r) { return r.json(); }).then(function(resp) {
        if (resp.ok) {
          window.AgToast?.show('Document ajoute', 'success');
          // Refresh doc badge count on this card
          updateOperatorDocBadge(cardEl, motionId);
        } else {
          window.AgToast?.show(resp.error || 'Erreur', 'error');
        }
      }).catch(function() {
        window.AgToast?.show('Erreur reseau', 'error');
      });
    });
    input.click();
  });

  // Add badge + button to card
  var actionsArea = cardEl.querySelector('.motion-actions') || cardEl;
  actionsArea.appendChild(uploadBtn);
}

function updateOperatorDocBadge(cardEl, motionId) {
  window.api('/api/v1/resolution_documents?motion_id=' + motionId).then(function(resp) {
    var count = resp.documents ? resp.documents.length : 0;
    var badge = cardEl.querySelector('.doc-badge') || document.createElement('span');
    badge.className = 'doc-badge' + (count > 0 ? ' doc-badge--has-docs' : ' doc-badge--empty');
    badge.textContent = count > 0 ? count + ' doc' + (count > 1 ? 's' : '') : 'Aucun doc';
    if (!badge.parentElement) cardEl.appendChild(badge);
  });
}
```

Wire `addDocUploadToMotionCard()` into the motion card rendering loop in the operator motions tab.
