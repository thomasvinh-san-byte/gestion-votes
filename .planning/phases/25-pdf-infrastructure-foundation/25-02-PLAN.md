**1. Create `public/assets/js/components/ag-pdf-viewer.js`:**

Follow the `AgModal` pattern exactly. Class `AgPdfViewer extends HTMLElement` with Shadow DOM.

Observed attributes: `['src', 'filename', 'mode', 'open', 'allow-download']`

- `mode="inline"` — embedded in page flow (desktop side panel). CSS: `display: block; width: 100%; height: var(--pdf-viewer-height, 600px); border-radius: var(--radius); overflow: hidden;`
- `mode="sheet"` — bottom sheet overlay (mobile voter). CSS: `position: fixed; inset: auto 0 0 0; height: 80dvh; background: var(--color-surface); border-radius: var(--radius-lg) var(--radius-lg) 0 0; z-index: var(--z-modal); transform: translateY(100%); transition: transform var(--duration-slow, 0.3s) var(--ease-out, ease-out); box-shadow: 0 -4px 24px rgba(0,0,0,0.18);`
- `mode="panel"` — side panel from right (desktop hub/operator). CSS: `position: fixed; inset: 0 0 0 auto; width: min(480px, 90vw); background: var(--color-surface); z-index: var(--z-modal); transform: translateX(100%); transition: transform var(--duration-slow, 0.3s) var(--ease-out, ease-out); box-shadow: -4px 0 24px rgba(0,0,0,0.18);`

When `open` attribute is set/removed, add/remove `[open]` which CSS transitions use.

Methods:
- `open()` — set `this.setAttribute('open', '')`; if sheet/panel mode, add backdrop click listener to close; dispatch `CustomEvent('ag-pdf-viewer-open', { bubbles: true })`
- `close()` — remove `open` attribute; clear iframe src to stop loading; dispatch `CustomEvent('ag-pdf-viewer-close', { bubbles: true })`
- `render()` — rebuild shadow DOM from template
- `attributeChangedCallback(name, old, val)` — if `src` changed, update iframe src to the new URL; if `filename` changed, update h3 text; if `open` changed, handle transitions; if `allow-download` present, show download button
- Private `_onBackdropClick()` — close on backdrop click
- Private `_onKeyDown(e)` — close on Escape key

Download button: Only visible when `allow-download` attribute is present. Links to the serve URL with `download` behavior. Voter mode will NOT set this attribute (PDF-10 requirement).

`escapeHtml(s)` instance method following the existing Web Component pattern for safe text insertion.

**FilePond setup function:**
```javascript
function initResolutionPond(containerEl, motionId, meetingId) {
  var inputEl = containerEl.querySelector('.filepond-input');
  if (!inputEl || inputEl._pondInitialized) return null;

  FilePond.registerPlugin(
    FilePondPluginFileValidateType,
    FilePondPluginFileValidateSize
  );

  var pond = FilePond.create(inputEl, {
    acceptedFileTypes: ['application/pdf'],
    labelFileTypeNotAllowed: 'Seuls les fichiers PDF sont acceptes',
    fileValidateTypeLabelExpectedTypes: 'Format attendu : PDF',
    maxFileSize: '10MB',
    labelMaxFileSizeExceeded: 'Le fichier depasse 10 Mo',
    labelMaxFileSize: 'Taille maximale : 10 Mo',
    allowMultiple: true,
    server: {
      process: {
        url: '/api/v1/resolution_documents',
        method: 'POST',
        headers: function() {
          return {
            'X-Requested-With': 'XMLHttpRequest',
            'X-CSRF-Token': document.querySelector('meta[name="csrf-token"]')?.content || ''
          };
        },
        ondata: function(formData) {
          formData.append('motion_id', motionId);
          formData.append('meeting_id', meetingId);
          return formData;
        },
        onload: function(response) {
          var data = JSON.parse(response);
          renderDocCard(containerEl.querySelector('.doc-list'), data.document);
          return data.document.id;
        },
        onerror: function(response) {
          try {
            var data = JSON.parse(response);
            window.AgToast?.show(data.error || 'Erreur lors du telechargement', 'error');
          } catch(e) {
            window.AgToast?.show('Erreur lors du telechargement', 'error');
          }
        }
      },
      revert: null  // use custom delete via doc cards, not FilePond revert
    },
    labelIdle: 'Glissez un PDF ici ou <span class="filepond--label-action">parcourir</span>',
  });

  inputEl._pondInitialized = true;
  return pond;
}
```

**Document card rendering function:**
```javascript
function renderDocCard(listEl, doc) {
  var card = document.createElement('div');
  card.className = 'doc-card';
  card.dataset.docId = doc.id;

  var sizeKb = Math.round(doc.file_size / 1024);
  var sizeLabel = sizeKb > 1024 ? (sizeKb / 1024).toFixed(1) + ' Mo' : sizeKb + ' Ko';

  card.innerHTML =
    '<span class="doc-card__icon">&#128196;</span>' +
    '<span class="doc-card__name">' + escapeHtml(doc.original_name) + '</span>' +
    '<span class="doc-card__size">' + escapeHtml(sizeLabel) + '</span>' +
    '<button class="doc-card__preview btn btn--ghost btn--sm" title="Apercu">' +
      '<span class="icon">&#128065;</span>' +
    '</button>' +
    '<button class="doc-card__delete btn btn--ghost btn--sm btn--danger" title="Supprimer">' +
      '<span class="icon">&times;</span>' +
    '</button>';

  // Preview button — open ag-pdf-viewer in panel mode
  card.querySelector('.doc-card__preview').addEventListener('click', function() {
    var viewer = document.querySelector('ag-pdf-viewer') || document.createElement('ag-pdf-viewer');
    if (!viewer.parentElement) {
      viewer.setAttribute('mode', 'panel');
      viewer.setAttribute('allow-download', '');
      document.body.appendChild(viewer);
    }
    viewer.setAttribute('src', '/api/v1/resolution_document_serve?id=' + doc.id);
    viewer.setAttribute('filename', doc.original_name);
    viewer.open();
  });

  // Delete button — use ag-confirm dialog
  card.querySelector('.doc-card__delete').addEventListener('click', function() {
    window.AgConfirm.ask({
      title: 'Supprimer le document',
      message: 'Voulez-vous supprimer "' + escapeHtml(doc.original_name) + '" ?',
      confirmLabel: 'Supprimer',
      danger: true
    }).then(function(confirmed) {
      if (!confirmed) return;
      window.api('/api/v1/resolution_documents', { id: doc.id }, 'DELETE').then(function() {
        card.remove();
        window.AgToast?.show('Document supprime', 'success');
      }).catch(function() {
        window.AgToast?.show('Erreur lors de la suppression', 'error');
      });
    });
  });

  listEl.appendChild(card);
}
```

**Load existing documents when editing a resolution:**
```javascript
function loadExistingDocs(containerEl, motionId, tenantId) {
  var listEl = containerEl.querySelector('.doc-list');
  if (!listEl) return;
  window.api('/api/v1/resolution_documents?motion_id=' + motionId).then(function(resp) {
    if (resp && resp.documents) {
      resp.documents.forEach(function(doc) {
        renderDocCard(listEl, doc);
      });
    }
  });
}
```
