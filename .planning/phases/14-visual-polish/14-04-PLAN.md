**Reference code pattern:**
```javascript
test('POLISH-04: .btn-primary has interactive hover state @critical-path', async ({ page }) => {
  // Login and navigate — reuse existing helper and a page already used in this spec
  await loginAsAdmin(page); // or whatever helper the spec already imports
  await page.goto('/meetings.htmx.html', { waitUntil: 'domcontentloaded' });

  const btn = page.locator('.btn-primary').first();
  await expect(btn).toBeVisible({ timeout: 10000 });

  const beforeStyles = await btn.evaluate((el) => {
    const cs = window.getComputedStyle(el);
    return { transform: cs.transform, boxShadow: cs.boxShadow };
  });

  await btn.hover();
  // Give the transition a chance to apply
  await page.waitForTimeout(250);

  const afterStyles = await btn.evaluate((el) => {
    const cs = window.getComputedStyle(el);
    return { transform: cs.transform, boxShadow: cs.boxShadow };
  });

  const changed =
    beforeStyles.transform !== afterStyles.transform ||
    beforeStyles.boxShadow !== afterStyles.boxShadow;
  expect(changed, `btn-primary hover must change transform or box-shadow (before=${JSON.stringify(beforeStyles)}, after=${JSON.stringify(afterStyles)})`).toBe(true);
});
```
