```css
/* ========================================================================== */
/* @media print — EDITORIAL-07                                                */
/* Pages éditoriales imprimées : masquer UI, page-break sur blocs critiques,  */
/* en-tête répété (titre + date), footer avec numéro de page.                 */
/* ========================================================================== */

@page {
  size: A4;
  margin: 2cm 1.5cm 2cm 1.5cm;

  @bottom-right {
    content: "Page " counter(page) " sur " counter(pages);
    font-family: var(--font-sans);
    font-size: 9pt;
    color: #555;
  }
}

/* En-tête print : caché hors impression (F-5 fix). Visible uniquement en @media print. */
.ag-editorial-print-header {
  display: none;
}

@media print {
  /* Réinitialiser couleurs / fonds pour économiser l'encre et garantir la lisibilité N&B */
  :root,
  body {
    background: #fff !important;
    color: #000 !important;
  }

  /* Wrapper imprimé : pas de grid sidebar, flux vertical pleine largeur */
  .ag-editorial {
    display: block;
    max-width: 100%;
    margin: 0;
    padding: 0;
    font-size: 11pt;
    line-height: 1.5;
  }

  .ag-editorial > .ag-editorial-content,
  .ag-editorial > main,
  .ag-editorial > article {
    max-width: 100%;
  }

  /* Masquer les contrôles UI : boutons, filtres, sidebar de navigation, modals */
  .ag-editorial .btn,
  .ag-editorial button,
  .ag-editorial input,
  .ag-editorial select,
  .ag-editorial .filter-tab,
  .ag-editorial .audit-filter-tabs,
  .ag-editorial .audit-filter-tabs__more,
  .ag-editorial details > summary,
  .ag-editorial > .ag-editorial-sidebar,
  .ag-editorial > aside,
  ag-integrity-modal,
  .ag-integrity-footer a,
  nav,
  header.app-header,
  .htmx-indicator {
    display: none !important;
  }

  /* Conserver visible le hash d'intégrité dans le footer du PV (sans le lien) */
  .ag-integrity-footer {
    display: block !important;
    margin-top: 1cm;
    border-top: 1px solid #000;
    padding-top: 0.5cm;
    font-size: 9pt;
    font-family: var(--font-mono);
    page-break-before: avoid;
  }

  /* Blocs critiques : éviter les sauts de page au milieu */
  .ag-resolution,
  .ag-resolution-pill,
  .ag-integrity-footer,
  .ag-editorial table,
  .ag-editorial blockquote,
  .ag-editorial pre {
    page-break-inside: avoid;
    break-inside: avoid;
  }

  /* En-têtes : pas de saut juste après */
  .ag-editorial h1,
  .ag-editorial h2,
  .ag-editorial h3 {
    page-break-after: avoid;
    break-after: avoid;
  }

  /* Liens : afficher l'URL après le texte sauf intra-document */
  .ag-editorial a[href]:not([href^="#"]):after {
    content: " (" attr(href) ")";
    font-size: 9pt;
    font-family: var(--font-mono);
    color: #555;
  }

  /* En-tête (F-5 fix Schoger) : la spec CSS Paged Media `position: running()` +
     `@page { @top-left { content: element(...) } }` n'est supportée que par Prince
     et antiword — IGNORÉE par Chrome / Firefox / Safari à l'impression navigateur.
     Pragmatic fallback : un <header class="ag-editorial-print-header"> caché par
     défaut, affiché en print uniquement, qui apparaît au début du document.
     L'en-tête répété sur chaque page n'est PAS livré par Phase 2 — accepté
     comme limitation. EDITORIAL-07 partial : "en-tête répété" devient "en-tête
     présent en début de document imprimé". Si en-tête sur chaque page est requis,
     route via dompdf (ProcurationPdfService déjà présent) en backlog v2.4. */
  .ag-editorial-print-header {
    display: block;  /* visible en print (caché hors print plus haut dans le fichier) */
    font-family: var(--font-sans);
    font-size: 9pt;
    color: #555;
    border-bottom: 1px solid #000;
    padding-bottom: 0.3cm;
    margin-bottom: 0.5cm;
  }

  /* Hashes/UUID en monospace lisible */
  .ag-mono,
  code,
  .ag-resolution-pill {
    font-family: var(--font-mono);
    color: #000;
  }
}
```
