    **Part A: Unsaved changes warning on wizard (D-12)**

    1. In wizard.js, add dirty state tracking. After the wizard form is loaded/rendered:
    ```js
    var _wizardSnapshot = null;
    var _wizardDirty = false;

    function captureWizardSnapshot() {
      var form = document.querySelector('.wizard-form') || document.querySelector('.wizard-step.active');
      if (!form) return;
      _wizardSnapshot = new FormData(form);
      _wizardDirty = false;
    }

    function isWizardDirty() {
      if (_wizardDirty) return true;
      var form = document.querySelector('.wizard-form') || document.querySelector('.wizard-step.active');
      if (!form || !_wizardSnapshot) return false;
      var current = new FormData(form);
      for (var [key, val] of current) {
        if (_wizardSnapshot.get(key) !== val) return true;
      }
      return false;
    }
    ```

    2. Listen for input changes to set dirty flag:
    ```js
    document.addEventListener('input', function(e) {
      if (e.target.closest('.wizard-step')) _wizardDirty = true;
    }, true);
    ```

    3. Add `beforeunload` handler:
    ```js
    window.addEventListener('beforeunload', function(e) {
      if (isWizardDirty()) {
        e.preventDefault();
        e.returnValue = '';
      }
    });
    ```

    4. For shell navigation (in-app page transitions that don't trigger beforeunload), check if `window.Shell` has a navigation intercept hook. If Shell.beforeNavigate exists, register:
