Since EmailQueueRepository extends AbstractRepository (needs PDO), we cannot unit-test the actual SQL execution. Instead, create a focused test that:

2. **Verifies markFailed SQL contains permanent failure logic**: Assert the SQL contains `WHEN retry_count + 1 >= max_retries THEN 'failed'`.

3. **Verifies fetchPendingBatch SQL respects retry_count**: Assert the SQL contains `retry_count < max_retries`.

4. **Verifies the email_queue table schema has retry columns**: Read `database/migrations/005_email_system.sql` and assert it contains `retry_count INT DEFAULT 0` and `max_retries INT DEFAULT 3`.

Implementation: Use `ReflectionMethod` to read the source of markFailed() and fetchPendingBatch(), or simply use `file_get_contents()` on the PHP source file to check for the SQL patterns. The file_get_contents approach is simpler and more reliable.

```php
class EmailQueueRepositoryRetryTest extends TestCase {
    private string $repoSource;
    private string $migrationSource;

    protected function setUp(): void {
        $this->repoSource = file_get_contents(__DIR__ . '/../../app/Repository/EmailQueueRepository.php');
        $this->migrationSource = file_get_contents(__DIR__ . '/../../database/migrations/005_email_system.sql');
    }

    public function testMarkFailedUsesExponentialBackoff(): void {
        $this->assertStringContainsString("power(2, retry_count)", $this->repoSource);
    }

    public function testMarkFailedSetsPermanentFailureAtMaxRetries(): void {
        $this->assertStringContainsString("retry_count + 1 >= max_retries THEN 'failed'", $this->repoSource);
    }

    public function testFetchPendingBatchRespectsRetryCount(): void {
        $this->assertStringContainsString("retry_count < max_retries", $this->repoSource);
    }

    public function testMigrationDefinesRetryColumns(): void {
        $this->assertStringContainsString("retry_count INT DEFAULT 0", $this->migrationSource);
        $this->assertStringContainsString("max_retries INT DEFAULT 3", $this->migrationSource);
    }

    public function testFetchPendingBatchUsesSkipLocked(): void {
        $this->assertStringContainsString("FOR UPDATE SKIP LOCKED", $this->repoSource);
    }

    public function testMarkFailedReschedulesWithInterval(): void {
        // Backoff formula: 5 minutes * 2^retry_count
        $this->assertStringContainsString("interval '5 minutes' * power(2, retry_count)", $this->repoSource);
    }
}
```
