<!-- API contracts from research (verified against routes.php and controllers) -->

wizard_status response (GET /api/v1/wizard_status?meeting_id=...):
```json
{
  "ok": true,
  "data": {
    "meeting_id": "uuid",
    "meeting_title": "string",
    "meeting_status": "draft|scheduled|frozen|live|paused|closed|validated|archived",
    "current_motion_id": "uuid|null",
    "members_count": 42,
    "present_count": 18,
    "motions_total": 5,
    "motions_closed": 3,
    "has_president": true,
    "quorum_met": false,
    "policies_assigned": true
  }
}
```

meeting_workflow_check response (GET /api/v1/meeting_workflow_check?meeting_id=...):
```json
{
  "current_status": "scheduled",
  "transitions": {
    "frozen": { "can_proceed": false, "issues": [{"code": "no_attendance", "msg": "..."}], "warnings": [] },
    "draft":  { "can_proceed": true,  "issues": [], "warnings": [] }
  }
}
```

invitations_stats response (GET /api/v1/invitations_stats?meeting_id=...):
```json
{
  "stats": { "pending": 5, "sent": 37 }
}
```

invitations_send_bulk (POST /api/v1/invitations_send_bulk):
body: { "meeting_id": "uuid", "only_unsent": true }
response: { "ok": true, "data": { "sent": 12 } }

meeting_transition (POST /api/v1/meeting_transition):
body: { "meeting_id": "uuid", "to_status": "frozen" }
response: { "ok": true } or { "error": "workflow_issues", "data": { "issues": [...] } }

<!-- Checklist item to data source mapping -->
convocation sent = invitations_stats.stats.sent > 0 (or stats.pending === 0)
quorum reached = wizard_status.data.quorum_met === true
agenda locked = meeting_status === 'frozen' || meeting_status === 'live'
