```javascript
// @ts-check
const { test, expect } = require('@playwright/test');

/**
 * MODAL-02 : Focus trap a11y des modales <ag-modal>.
 *
 * Vérifie le contrat critique :
 *  1. Escape ferme la modale
 *  2. Le focus est restoré sur l'élément déclencheur après fermeture
 *  3. Tab cycle à l'intérieur de la modale (focus trap)
 *
 * Cible : une page parmi les 6 migrées par le plan 04.2.
 * À adapter : chemin de page + sélecteur du trigger en fonction du SUMMARY 04.2.
 */

const PAGE_PATH = '/operator.htmx.html'; // À ajuster si nécessaire d'après SUMMARY 04.2

test.describe('MODAL-02 — focus trap <ag-modal>', () => {
  test('Escape ferme la modale et restore le focus au trigger', async ({ page }) => {
    await page.goto(PAGE_PATH);

    // Trouver un trigger qui ouvre une <ag-modal>
    const trigger = page.locator('[aria-haspopup="dialog"]').first();
    await expect(trigger).toBeVisible({ timeout: 5000 });

    // Mémoriser le trigger pour vérifier la restauration de focus
    await trigger.focus();
    await expect(trigger).toBeFocused();

    // Ouvrir la modale (clic ou Enter)
    await trigger.click();

    // La <ag-modal> doit être visible et ouverte
    const modal = page.locator('ag-modal[open], ag-modal:not([hidden])').first();
    await expect(modal).toBeVisible({ timeout: 2000 });

    // Le focus doit avoir été déplacé à l'intérieur de la modale (focus trap initial)
    const focusInsideModal = await page.evaluate(() => {
      const modal = document.querySelector('ag-modal');
      if (!modal) return false;
      const active = document.activeElement;
      return modal.contains(active) || (modal.shadowRoot && modal.shadowRoot.contains(active));
    });
    expect(focusInsideModal).toBe(true);

    // Appuyer Escape doit fermer la modale
    await page.keyboard.press('Escape');
    await expect(modal).toBeHidden({ timeout: 2000 });

    // Le focus doit être restoré sur le trigger
    await expect(trigger).toBeFocused();
  });

  test('Tab cycle à l\'intérieur de la modale (focus trap)', async ({ page }) => {
    await page.goto(PAGE_PATH);

    const trigger = page.locator('[aria-haspopup="dialog"]').first();
    await trigger.click();

    const modal = page.locator('ag-modal[open], ag-modal:not([hidden])').first();
    await expect(modal).toBeVisible({ timeout: 2000 });

    // Tab plusieurs fois — le focus ne doit jamais sortir de la modale
    for (let i = 0; i < 10; i++) {
      await page.keyboard.press('Tab');
      const stillInside = await page.evaluate(() => {
        const modal = document.querySelector('ag-modal');
        if (!modal) return false;
        const active = document.activeElement;
        return modal.contains(active) || (modal.shadowRoot && modal.shadowRoot.contains(active));
      });
      expect(stillInside).toBe(true);
    }

    // Cleanup
    await page.keyboard.press('Escape');
  });
});
```
