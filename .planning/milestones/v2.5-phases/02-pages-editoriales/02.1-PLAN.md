```css
/*
 * editorial.css — Pages éditoriales (audit, trust, archives, report)
 * Phase 2 / EDITORIAL-01..09
 *
 * Wrapper .ag-editorial : Fraunces serif sur contenu, max-width 720px,
 * grid layout avec sidebar ≥1024px. Pas de text-align: center sur enfants directs.
 */

/* ---- Wrapper principal : grid 2 colonnes (contenu + sidebar) ---- */
.ag-editorial {
  display: grid;
  grid-template-columns: minmax(0, 720px) minmax(0, 1fr);
  gap: var(--space-6); /* ajuster au token réel ; ex: --gap-lg */
  max-width: 1100px;
  margin-inline: auto;
  padding-inline: var(--space-4);
  font-family: var(--font-display);
  font-size: var(--font-size-base, 1rem);
  line-height: 1.6;
  color: var(--text-primary, inherit);
}

/* ---- Verrou text-align: left sur tous les enfants directs ---- */
/* Refactoring UI : centrer du long-form text est un anti-pattern de lisibilité. */
.ag-editorial > * {
  text-align: left;
}

/* ---- Colonne contenu (première grid cell) ---- */
.ag-editorial > .ag-editorial-content,
.ag-editorial > main,
.ag-editorial > article {
  max-width: 720px;
  line-height: 1.55;
}

/* ---- Sidebar (deuxième grid cell — hash, méta, nav interne) ---- */
.ag-editorial > .ag-editorial-sidebar,
.ag-editorial > aside {
  font-family: var(--font-sans);
  font-size: var(--font-size-sm, 0.875rem);
  line-height: 1.5;
}

/* ---- Contrôles UI : forcer var(--font-sans) (Bricolage) ---- */
.ag-editorial .btn,
.ag-editorial button,
.ag-editorial input,
.ag-editorial select,
.ag-editorial .filter-tab,
.ag-editorial .audit-filter-tabs,
.ag-editorial details > summary {
  font-family: var(--font-sans);
}

/* ---- Hashes / UUID / codes : JetBrains Mono ---- */
.ag-editorial code,
.ag-editorial .ag-mono,
.ag-editorial .hash,
.ag-editorial .uuid {
  font-family: var(--font-mono);
  font-size: 0.95em;
}

/* ---- Pill résolution monospace (en-tête / liste / tableau uniquement) ---- */
/* EDITORIAL-04 : interdit inline en flux paragraphe — appliquée par convention HTML */
.ag-resolution-pill {
  display: inline-flex;
  align-items: center;
  font-family: var(--font-mono);
  font-size: 0.875em;
  padding-inline: var(--space-2);
  padding-block: var(--space-1);
  border-radius: var(--radius-pill);
  background: var(--surface-sunken, #f3f4f6);
  border: 1px solid var(--border-subtle);
  white-space: nowrap;
}

/* ---- Mobile : collapse grid en flux vertical ---- */
@media (max-width: 1023.98px) {
  .ag-editorial {
    grid-template-columns: minmax(0, 1fr);
  }
}

/* ---- Sous 768px : largeur 100% + padding latéral, pas de scroll horizontal ---- */
@media (max-width: 767.98px) {
  .ag-editorial {
    padding-inline: var(--space-3);
    gap: var(--space-4);
  }
  .ag-editorial > .ag-editorial-content,
  .ag-editorial > main,
  .ag-editorial > article {
    max-width: 100%;
  }
}
```
