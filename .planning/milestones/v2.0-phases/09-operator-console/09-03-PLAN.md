```javascript
function showQuorumWarning(stats) {
  // stats = { present, inscrits, requis }
  var confirm = document.createElement('ag-confirm');
  confirm.setAttribute('variant', 'warn');
  confirm.setAttribute('title', 'Quorum non atteint');
  confirm.setAttribute('no-backdrop-close', '');

  // Build custom body with stat cards + 3 action buttons
  var body = document.createElement('div');
  body.className = 'op-quorum-body';
  body.innerHTML =
    '<div class="op-quorum-stats">' +
      '<div class="op-quorum-stat"><span class="op-quorum-stat-value">' + stats.present + '</span><span class="op-quorum-stat-label">Presents</span></div>' +
      '<div class="op-quorum-stat"><span class="op-quorum-stat-value">' + stats.inscrits + '</span><span class="op-quorum-stat-label">Inscrits</span></div>' +
      '<div class="op-quorum-stat"><span class="op-quorum-stat-value">' + stats.requis + '</span><span class="op-quorum-stat-label">Requis</span></div>' +
    '</div>' +
    '<p class="op-quorum-warning-text">Le quorum n\'est pas atteint. Choisissez une action :</p>' +
    '<div class="op-quorum-actions">' +
      '<button class="btn btn-warning" data-quorum-action="reporter">Reporter (2e convocation)</button>' +
      '<button class="btn btn-secondary" data-quorum-action="suspendre">Suspendre 30 min</button>' +
      '<button class="btn btn-ghost" data-quorum-action="continuer">Continuer sous reserve</button>' +
    '</div>' +
    '<p class="op-quorum-risk-note text-xs text-muted" hidden id="opQuorumRiskNote">Attention : continuer sans quorum expose les decisions a un risque de contestation.</p>';

  confirm.appendChild(body);
  document.body.appendChild(confirm);

  // Handle actions
  body.querySelectorAll('[data-quorum-action]').forEach(function(btn) {
    btn.addEventListener('click', function() {
      var action = btn.dataset.quorumAction;
      if (action === 'continuer') {
        // Show risk warning first click, confirm on second
        var riskNote = document.getElementById('opQuorumRiskNote');
        if (riskNote && riskNote.hidden) {
          riskNote.hidden = false;
          btn.textContent = 'Confirmer : continuer';
          btn.classList.add('btn-danger');
          btn.classList.remove('btn-ghost');
          return;
        }
      }
      confirm.remove();
      handleQuorumAction(action);
    });
  });
}

function handleQuorumAction(action) {
  if (action === 'reporter') {
    // Mark meeting for 2nd convocation
    O.fn.srAnnounce('Seance reportee pour 2e convocation');
  } else if (action === 'suspendre') {
    // Start 30-min suspension timer
    O.fn.srAnnounce('Seance suspendue pour 30 minutes');
  } else if (action === 'continuer') {
    O.fn.srAnnounce('Seance continue sous reserve de quorum');
  }
}
```

**Keyboard shortcuts:**
```javascript
document.addEventListener('keydown', function(e) {
  // Skip if in input/textarea or meta/ctrl held
  var tag = document.activeElement ? document.activeElement.tagName.toLowerCase() : '';
  if (tag === 'input' || tag === 'textarea' || tag === 'select') return;
  if (e.metaKey || e.ctrlKey || e.altKey) return;

  if (e.key === 'p' || e.key === 'P') {
    e.preventDefault();
    var proclaimBtn = document.getElementById('opBtnProclaim');
    if (proclaimBtn && !proclaimBtn.disabled) handleProclaim();
  }
  if (e.key === 'f' || e.key === 'F') {
    e.preventDefault();
    var toggleBtn = document.getElementById('opBtnToggleVote');
    if (toggleBtn && !toggleBtn.disabled) toggleBtn.click();
  }
});
```

```javascript
function renderAgendaList() {
  var list = document.getElementById('opAgendaList');
  if (!list) return;

  list.innerHTML = O.motionsCache.map(function(m, i) {
    var status = m.closed_at ? 'voted' : (O.currentOpenMotion && O.currentOpenMotion.id === m.id ? 'current' : 'pending');
    return '<div class="op-agenda-item ' + status + '" data-motion-id="' + escapeHtml(m.id) + '" role="button" tabindex="0">' +
      '<span class="op-agenda-circle"></span>' +
      '<span class="op-agenda-title">' + (i + 1) + '. ' + escapeHtml(m.title) + '</span>' +
    '</div>';
  }).join('');

  // Click handler: navigate to resolution, reset to Resultat tab
  list.querySelectorAll('.op-agenda-item[data-motion-id]').forEach(function(item) {
    item.addEventListener('click', function() {
      var motionId = item.dataset.motionId;
      O.fn.selectMotion(motionId);
      // Reset to Resultat sub-tab
      var tabs = document.querySelectorAll('.op-tab');
      var panels = document.querySelectorAll('.op-tab-panel');
      tabs.forEach(function(t) { t.classList.remove('active'); });
      panels.forEach(function(p) { p.classList.remove('active'); });
      var resultatTab = document.querySelector('[data-op-tab="resultat"]');
      var resultatPanel = document.getElementById('opPanelResultat');
      if (resultatTab) resultatTab.classList.add('active');
      if (resultatPanel) resultatPanel.classList.add('active');
    });
  });
}
```
