  var lieu = [place, addr].filter(Boolean).join(', ') || '(non renseign\u00e9)';

    warnings += '<div class="review-warning">\u26a0 Aucun membre ajout\u00e9 \u2014 les votes ne pourront pas \u00eatre attribu\u00e9s</div>';
