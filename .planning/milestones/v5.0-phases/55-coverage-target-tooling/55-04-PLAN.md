```php
class ControllerTestCase extends TestCase {
    protected function injectRepos(array $repoMocks): void;  // Injects mocks into RepositoryFactory singleton
    protected function callController(string $class, string $method): array;  // Returns ['status', 'body']
    protected function injectJsonBody(array $data): void;
    protected function setHttpMethod(string $method): void;
    protected function setQueryParams(array $params): void;
    protected function setAuth(string $userId, string $role, string $tenantId): void;
}
```

Usage pattern:
```php
class SomeControllerTest extends ControllerTestCase {
    public function testEndpoint(): void {
        $meetingRepo = $this->createMock(MeetingRepository::class);
        $meetingRepo->method('findById')->willReturn([...]);
        $this->injectRepos([MeetingRepository::class => $meetingRepo]);

        $this->setHttpMethod('POST');
        $this->injectJsonBody(['meeting_id' => 'uuid-here']);

        $result = $this->callController(SomeController::class, 'methodName');
        $this->assertEquals(200, $result['status']);
    }
}
```
