**JS changes in settings.js:**

1. Variable tag click-to-insert: Add click handler on `.variable-tag` buttons that inserts the variable text at the textarea cursor position:
```javascript
document.addEventListener('click', function(e) {
  var tag = e.target.closest('.variable-tag');
  if (!tag) return;
  e.preventDefault();
  var varText = tag.dataset.var;
  var textarea = document.getElementById('templateBody');
  if (!textarea) return;
  var start = textarea.selectionStart;
  var end = textarea.selectionEnd;
  var text = textarea.value;
  textarea.value = text.substring(0, start) + varText + text.substring(end);
  textarea.selectionStart = textarea.selectionEnd = start + varText.length;
  textarea.focus();
  // Trigger preview update
  updateTemplatePreview();
});
```

2. Live preview update: Add a function that renders the template body with sample data into `#templatePreviewRender`:
```javascript
function updateTemplatePreview() {
  var body = document.getElementById('templateBody');
  var preview = document.getElementById('templatePreviewRender');
  if (!body || !preview) return;
  var text = body.value || '';
  // Replace variables with sample values for preview
  var samples = {
    '{{nom}}': 'Jean Dupont',
    '{{date}}': '15/04/2026',
    '{{heure}}': '14h00',
    '{{lieu}}': 'Salle du Conseil',

    '{{lien_vote}}': 'https://vote.example.com/abc123'
  };
  Object.keys(samples).forEach(function(k) {
    text = text.split(k).join('<strong>' + samples[k] + '</strong>');
  });

}
```
