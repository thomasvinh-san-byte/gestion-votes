      if (!r.body || !r.body.ok || !r.body.data) return;

      Object.keys(settings).forEach(function(key) {
        var el = document.getElementById(key);
        if (!el) return;
        var val = settings[key];
        if (el.type === 'checkbox') {
          el.checked = !!val;
          _prevValues.set(key, !!val);
        } else {
          el.value = val;
          _prevValues.set(key, val);
        }
      });

}
