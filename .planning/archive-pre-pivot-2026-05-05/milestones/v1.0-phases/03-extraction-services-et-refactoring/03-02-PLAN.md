    ```php
    public function membersCsv(): void {
        $in = api_request('POST');

        // File reading stays in controller (HTTP concern)
        // ... file upload / csv_content handling stays here ...

        $colIndex = ImportService::mapColumns($headers, ImportService::getMembersColumnMap());
        // Column validation stays in controller (HTTP response)
        $hasName = isset($colIndex['name']);
        $hasFirstLast = isset($colIndex['first_name']) && isset($colIndex['last_name']);
        if (!$hasName && !$hasFirstLast) {
            api_fail('missing_name_column', 400, [...]);
        }

        // Duplicate check — now throws InvalidArgumentException
        try {
            ImportService::checkDuplicateEmails($rows, $colIndex);
        } catch (\InvalidArgumentException $e) {
            api_fail('duplicate_emails', 400, ['detail' => $e->getMessage()]);
        }

        $tenantId = api_current_tenant_id();
        $result = ['imported' => 0, 'skipped' => 0, 'errors' => []];

        self::wrapApiCall(function () use (&$result, $rows, $colIndex, $hasName, $hasFirstLast, $tenantId) {
            api_transaction(function () use (&$result, $rows, $colIndex, $hasName, $hasFirstLast, $tenantId) {
                $result = $this->importService()->processMemberImport($rows, $colIndex, $hasName, $hasFirstLast, $tenantId);
            });
        }, 'import_failed');

        audit_log('members_import', 'member', null, [...]);
        api_ok(['imported' => $result['imported'], 'skipped' => $result['skipped'], 'errors' => array_slice($result['errors'], 0, 20)]);
    }
    ```
